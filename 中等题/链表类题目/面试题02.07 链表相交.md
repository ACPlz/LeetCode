class Solution {
public
    ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        int numberCountA = 0;
        int numberCountB = 0;
        int gap;
        ListNode curA = headA;
        ListNode curB = headB;
        while (curA != NULL) {
            curA = curA-next;
            numberCountA++;
        }
        while (curB != NULL) {
            curB = curB-next; 
            numberCountB++;
        }
        curA = headA;
        curB = headB;
        if (numberCountA  numberCountB) {
            gap = numberCountA - numberCountB;
            while (gap--) {
                curA = curA-next;
            }
            while (curA != NULL) {
                if (curA == curB) {
                    return curA;
                }
                else {
                    curA = curA-next;
                    curB = curB-next;
                }
            }
            return NULL;
        }
        else {
            gap = numberCountB - numberCountA;
            while (gap--) {
                curB = curB-next;
            }
            while (curB != NULL) {
                if (curB == curA) {
                    return curB;
                }
                else {
                    curA = curA-next;
                    curB = curB-next;
                }
            }
            return NULL;            
        }   
    }
};
