class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int low = 0, high = nums.size() - 1;
        while (low <= high) {
            int mid = (high - low) / 2 + low;
            if (target > nums[mid]) {
                low = mid + 1;
            } else if (target < nums[mid]) {
                high = mid - 1;    
            } else return mid;
        }
        return low;
    }
};
//和二分查找一模一样就改了个return的值
