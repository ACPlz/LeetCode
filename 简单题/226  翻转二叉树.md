# 题目描述
## 翻转一棵二叉树。
### 我的题解：
```class Solution {
public:
    TreeNode* invertTree(TreeNode* root) {
        vector<int> result;
        stack<TreeNode*> st1;
        if (root == nullptr) return root;
        st1.push(root);
        while (!st1.empty()) {
            TreeNode* ans = st1.top();
            if (ans != nullptr) {
                st1.pop();
                st1.push(ans);
                st1.push(nullptr);
                if (ans->right) st1.push(ans->right);
                if (ans->left) st1.push(ans->left);
            }
            else {
                st1.pop();
                TreeNode* ans = st1.top();
                st1.pop();
                swap(ans->left, ans->right);
            }           
        }
        return root;
    }
};
```
### **备注**：遍历一遍，交换左右孩子即可，此处采用的是后序遍历
